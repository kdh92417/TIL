### 나의 틀림 풀이

```python
sheet = {
    "C#" : 'H',
    "D#" : 'I',
    "F#" : 'J',
    "G#" : 'K',
    "A#" : 'F'
}

def solution(m, musicinfos):
    
    answer = []
    melody, reesult = [], []
    
    for info in musicinfos:
        info = info.split(",")
        h1, m1 = info[0].split(":")
        h2, m2 = info[1].split(":")
        time = (int(h2)*60) + int(m2) - (int(h1)*60) + int(m1)
        s = []
        
        for mel in info[-1]:
            if mel.isalpha():
                s.append(mel)
            else:
                s[-1] = sheet[s[-1]+mel]
                
        mel = ''.join([s[i%len(s)] for i in range(time)])
        melody.append([time, mel, info[-2]])

    new_m = []
    for i in m:
        if i.isalpha():
            new_m.append(i)
        else:
            new_m[-1] = sheet[new_m[-1]+i]
    
    new_m = ''.join(new_m)
    for mel in melody:
        if new_m in mel[1]:
            answer.append([time, mel[-1]])
    
    answer.sort()
    
    return answer[-1][-1]
```
